{{- $covers := .Site.Data.covers -}} {{- /* Maintain a shuffled queue so covers only repeat after a full cycle. */ -}} {{- $scratchKey := "helpers/randomCover/queue" -}} {{- $queue := .Scratch.Get $scratchKey | default (slice) -}} {{- if lt (len $queue) 1 -}} {{- $queue = shuffle $covers -}} {{- end -}} {{- $cover := index $queue 0 -}} {{- $nextQueue := cond (gt (len $queue) 1) (after 1 $queue) (slice) -}} {{- .Scratch.Set $scratchKey $nextQueue -}} {{- return $cover -}}
